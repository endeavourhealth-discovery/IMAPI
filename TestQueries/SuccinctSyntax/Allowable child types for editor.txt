{  
  prefix im: http://endhealth.info/im#  
  prefix sh: http://www.w3.org/ns/shacl#  
  prefix rdfs: http://www.w3.org/2000/01/rdf-schema#  
  prefix sn: http://snomed.info/sct#
  arguments {
    this : @id im:Folder
    } 
  query {
    name "Allowable child types for editor"
    from {
      where {@id im:isContainedIn in {@id im:EntityTypes}}}
    select {@id rdfs:label},
      {@id sh:property
        where {@id sh:node in {@var $this}}
             and {@id sh:path in 
              {@id im:isContainedIn}
              or {@id rdfs:subClassOf}
              or {@id im:isSubsetOf}}
        select {@id sh:path}}}}
