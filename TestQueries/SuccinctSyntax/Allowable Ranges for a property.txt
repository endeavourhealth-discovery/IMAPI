{  
  prefix im: http://endhealth.info/im#  
  prefix sh: http://www.w3.org/ns/shacl#  
  prefix rdfs: http://www.w3.org/2000/01/rdf-schema#  
  prefix sn: http://snomed.info/sct#  
  prefix xsd: http://www.w3.org/2001/XMLSchema#
  arguments {
    this : @id im:Concept
    } 
  query {
    name "Allowable Ranges for a property"
    description "'using property domains get the allowable properties from the supertypes of this concept"
    activeOnly
    from {@type im:Concept
      where {inverseOf @id rdfs:range in {@var >><<$this}}}
    select {@id im:code},
      {@id rdfs:label}}}
